  var nombre=docroiValue('docroiNombre','[NOMBRE]');
  var colorPrincipal=docroiValue('docroiColorPrincipal','[COLOR PRINCIPAL]');
  var colorSecundario=docroiValue('docroiColorSecundario','[COLOR SECUNDARIO]');
  var contacto=docroiValue('docroiContacto','[DATOS DE CONTACTO]');
  var email=docroiValue('docroiEmail','[EMAIL RGPD]');
  var logo=docroiValue('docroiLogo','[LOGO]');
  var maps=docroiValue('docroiMaps','[URL GOOGLE MAPS]');
  var buyer=docroiValue('docroiBuyer','[BUYER PERSONA]');
  var prompt=docroiBasePrompt
    .replaceAll('[NOMBRE]',nombre)
    .replaceAll('[COLOR PRINCIPAL]',colorPrincipal)
    .replaceAll('[COLOR SECUNDARIO]',colorSecundario)
    .replaceAll('[DATOS DE CONTACTO]',contacto)
    .replaceAll('[EMAIL RGPD]',email)
    .replaceAll('[URL GOOGLE MAPS]',maps)
    .replaceAll('[LOGO]',logo)
    .replaceAll('[BUYER PERSONA]',buyer);
  prompt='IMPORTANTE PARA ESTA EJECUCIÓN:\nAdemás de este prompt, adjuntaré la ficha de buyer persona y el logo si los tengo como archivos. Si hay adjuntos, analízalos y úsalos. Si el buyer persona ya está pegado en el prompt, úsalo como fuente principal junto con cualquier adjunto.\n\n'+prompt;
  var area=document.getElementById('docroiPromptArea');if(area)area.value=prompt;

function docroiSelectPrompt(){var area=document.getElementById('docroiPromptArea');if(!area)return;area.focus();area.select();docroiSetStatus('Prompt seleccionado. Ahora copia con Ctrl+C o Cmd+C.');}
